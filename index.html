function printar(){ let html = ""; document.querySelectorAll(".card").forEach((c, i) => { if (c.querySelector(".nome")) { const nome = c.querySelector(".nome").value; const status = c.querySelector(".status").value; const obs = c.querySelector(".obs").value; const tecnico = c.querySelector(".tecnico").value; const analista = c.querySelector(".analista").value; const img = c.querySelector("img"); let imagemHTML = ""; if (img && img.src) { imagemHTML = ''; } html += `

Equipamento ${i}

Nome: ${nome}

Status: ${status}

Obs: ${obs}

 ${imagemHTML}

Técnico: ${tecnico}

Analista: ${analista}
`; } }); const w = window.open("", "_blank"); const pagina = `

Relatório Técnico

Projeto: ${document.getElementById("projeto").value}

Loja: ${document.getElementById("loja").value}

Responsável: ${document.getElementById("responsavel").value}

Data: ${document.getElementById("data").value}

Início: ${document.getElementById("inicio").value}

Término: ${document.getElementById("fim").value}

${html}
`; w.document.open(); w.document.write(pagina); w.document.close(); setTimeout(() => w.print(), 500); }
